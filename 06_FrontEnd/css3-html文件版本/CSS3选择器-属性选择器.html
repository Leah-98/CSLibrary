CSS3选择器-属性选择器

element[attribute]
     为带有attribute属性的element元素设置样式
     IE8+ FireFox Chrome Safari Opera

a[href] {
text-decoration:none;
}
<a href="attribute.html">attribute</a>

element[attribute="value"]
     为attribute="value"属性的Element元素设置样式

a[href="#"] {
color:#f00;
}
<a href="attribute.html">我不会变红</a>
<a href="#">我会变红</a>

Element[attribute~="value"]
     选择attribute属性包含单词"value"的元素并设置样式


a[class~="two"] {
color:#f00;
}
<a href="attribute.html">我不会变红</a>
<a class="one two">我会变红</a>
<a class="two">我会变红</a>
<a class="two three">我会变红</a>
<a href="twoo ">我不会变红</a>

Element[attribute^="value"]
     设置attribute属性以"value"开头的所有element元素的样式

a[href^="#"] {
color:#f00;
}
<a href="attribute.html">我不会变红</a>
<a href="#">我会变红</a>
<a href="#1">我会变红</a>

Element[attribute$="value"]
     设置attribute属性值以"value"结尾的所有element元素的样式

a[href$="#"] {
color:#f00;
}
<a href="attribute.html">我不会变红</a>
<a href="#">我会变红</a>
<a href="#1">我不会变红</a>
<a href="1#">我会变红</a>

Element[attribute*="value"]
     设置attribute属性值包含"value"的所有element元素的样式

a[href$="#"] {
color:#f00;
}
<a href="attribute.html">我不会变红</a>
<a href="#">我会变红</a>
<a href="#1">我会变红</a>
<a href="1#">我会变红</a>
<a href="1#1">我会变红</a>

element[attribute|="value"]
     选择attribute属性值为value或以"value-"开头的元素并设置其样式

a[href｜="#"] {
color:#f00;
}
<a href="attribute.html">我不会变红</a>
<a href="#">我会变红</a>
<a href="#1">我不会变红</a>
<a href="1#">我不会变红</a>
<a href="1#1">我不会变红</a>
<a href="#-1">我会变红</a>
